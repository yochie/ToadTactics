void PlayGame(){
    StartGame();
    this.CharacterDraftPhase();
    int winnerID = this.PlayRounds();
    EndGame(winnerID);
}

void CharacterDraftPhase(){
    int lastPlayerWhoDrafted = random.range(0, this.players.Length);
    while(!this.AllCharactersDrafted()){
        lastPlayerWhoDrafted = this.Draft(lastPlayerWhoDrafted);
    }
}

int PlayRounds(){
    int lastWinningPlayerID = random.range(0,this.players.Length);
    bool gameOver = false;
    while(!gameOver){
        lastWinningPlayerID = this.PlayRound(lastWinningPlayerID);
        this.gameScore[lastWinningPlayerID]++;
        gameOver = (this.gameScore[0] >= 2 || this.gameScore[1] >= 2);
        if(!gameOver){
            this.TreasureDraftPhase(lastWinningPlayerID);
            this.TreasureAssignmentPhase();        
        }
    }
    return lastWinningPlayerID;
}

int PlayRound(){
    this.CharacterPlacementPhase();
    int roundWinner = this.GameplayPhase();
    return roundWinner;
}

bool AllPlayersPlacedTheirCharacters(){
    foreach(PlayerController p in players){
        if(!p.hasPlacedAllCharacters()){
            return false
        } 
    }
    return true;
}

bool AllCharactersDrafted(){
    foreach(PlayerController p in players){
        if(!p.hasDraftedAllChars()){
            return false
        } 
    }
    return true;
}

PlayerController.hasDraftedAllChars(int numCharsToDraft){
    if (this.playerCharacters.Length < numCharsToDraft)
        return false;
    else
        return true;
}
